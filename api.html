JavaScript (browser)
// Works directly in browser — no backend needed
async function removeBg(imageFile) {
  const form = new FormData();
  form.append('image_file', imageFile);
  form.append('size', 'auto');

  const res = await fetch('https://api.remove.bg/v1.0/removebg', {
    method: 'POST',
    headers: { 'X-API-Key': 'YOUR_API_KEY' },
    body: form
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.errors?.[0]?.title ?? `HTTP ${res.status}`);
  }

  const blob = await res.blob();
  return URL.createObjectURL(blob); // ready to use in <img src>
}

from pathlib import Path

def remove_bg(input_path: str, api_key: str) -> bytes:
    path = Path(input_path)
    with path.open('rb') as f:
        res = requests.post(
            'https://api.remove.bg/v1.0/removebg',

            data={'size': 'auto'},
            headers={'X-API-Key': 'YOUR_API_KEY'},

    return res.content  # PNG bytes

Node.js
Node.js with retry logic
const { fetch, FormData, Blob } = require('undici');
const fs = require('fs');
const path = require('path');

async function removeBg(filePath, retries = 3) {
  const form = new FormData();
  form.set('image_file', new Blob([fs.readFileSync(filePath)]));
  form.set('size', 'auto');

  const res = await fetch('https://api.remove.bg/v1.0/removebg', {
    method: 'POST',
    headers: { 'X-API-Key': 'YOUR_API_KEY' },
    body: form,
  });

  // Handle rate limit with automatic retry
  if (res.status === 429 && retries > 0) {
    const wait = (+res.headers.get('Retry-After') || 10) * 1000;
    console.log(`Rate limited — retrying in ${wait / 1000}s`);
    await new Promise(r => setTimeout(r, wait));
    return removeBg(filePath, retries - 1);
  }

  if (!res.ok) {
    const body = await res.text();
    throw new Error(`HTTP ${res.status}: ${body}`);
  }

  const buf = Buffer.from(await res.arrayBuffer());
  const outName = path.basename(filePath, path.extname(filePath)) + '-cutout.png';
  fs.writeFileSync(outName, buf);
  console.log(`Saved: ${outName}`);
  return buf;
}
React Hook
React — drop-in hook
import { useState, useCallback } from 'react';

/**
 * useRemoveBg — drop this into any React component.
 * Pass your API key as a prop or env var.
 */
export function useRemoveBg(apiKey) {
  const [state, setState] = useState({
    loading: false,
    resultUrl: null,
    resultBlob: null,
    error: null,
  });

  const remove = useCallback(async (file) => {
    setState(s => ({ ...s, loading: true, error: null }));

    const form = new FormData();
    form.append('image_file', file);
    form.append('size', 'auto');

    try {
      const res = await fetch('https://api.remove.bg/v1.0/removebg', {
        method: 'POST',
        headers: { 'X-API-Key': apiKey },
        body: form,
      });
      if (!res.ok) {
        const d = await res.json().catch(() => ({}));
        throw new Error(d.errors?.[0]?.title ?? `Error ${res.status}`);
      }
      const blob = await res.blob();
      const url = URL.createObjectURL(blob);
      setState({ loading: false, resultUrl: url, resultBlob: blob, error: null });
    } catch (err) {
      setState({ loading: false, resultUrl: null, resultBlob: null, error: err.message });
    }
  }, [apiKey]);

  const reset = useCallback(() => {
    setState({ loading: false, resultUrl: null, resultBlob: null, error: null });
  }, []);

  return { ...state, remove, reset };
}

Rate Limits

Rate limit is resolution-dependent: 500 megapixel-images per minute. A 4 MP image counts as 4 units.

ResolutionEffective limit
625×400 (1 MP)500/min
1600×1200 (2 MP)250/min
2500×1600 (4 MP)125/min
4000×2500 (10 MP)50/min
6250×4000 (25 MP)20/min

On 429, read Retry-After header and wait that many seconds. On 5xx, use exponential backoff with jitter (1s, 2s, 4s, 8s).

Error Handling

All error responses are JSON with an errors array. Each error has a title and optional code.

Robust fetch wrapper
async function removeBgSafe(file, apiKey, retries = 3) {
  const form = new FormData();
  form.append('image_file', file);
  form.append('size', 'auto');

  let attempt = 0;

  while (attempt <= retries) {
    try {
      const res = await fetch('https://api.remove.bg/v1.0/removebg', {
        method: 'POST',
        headers: { 'X-API-Key': apiKey },
        body: form,
      });

      if (res.status === 429) { // rate limit — respect Retry-After
        const wait = +res.headers.get('Retry-After') * 1000 || 10000;
        await sleep(wait); attempt++; continue;
      }

      if (res.status >= 500) { // server error — exponential backoff + jitter
        const wait = (2 ** attempt) * 1000 + Math.random() * 500;
        await sleep(wait); attempt++; continue;
      }

      if (!res.ok) { // 4xx — don't retry
        const err = await res.json().catch(() => ({}));
        throw new Error(err.errors?.[0]?.title ?? `HTTP ${res.status}`);
      }

      return await res.blob(); // success ✓

    } catch (e) {
      if (attempt >= retries) throw e;
      await sleep(1500); attempt++;
    }
  }
}

const sleep = ms => new Promise(r => setTimeout(r, ms));
