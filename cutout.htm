async function removeBackground(imageFile) {
  const form = new FormData();
  form.append('image_file', imageFile);
  form.append('size', 'auto');

  const res = await fetch('https://api.remove.bg/v1.0/removebg', {
    method: 'POST',
    headers: { 'X-API-Key': 'YOUR_API_KEY' },
    body: form
  });

  if (!res.ok) throw new Error(await res.text());
  return res.blob(); // PNG blob, ready to display
}

Node.js — with retry 
const { FormData, fetch } = require('undici');

async function removeBackground(filePath, retries = 3) {
  const form = new FormData();
  form.set('image_file', new Blob([require('fs').readFileSync(filePath)]));
  form.set('size', 'auto');

  const res = await fetch('https://api.remove.bg/v1.0/removebg', {
    method: 'POST',
    headers: { 'X-API-Key': 'YOUR_API_KEY' },
    body: form
  });

  if (res.status === 429 && retries > 0) {
    const wait = +res.headers.get('Retry-After') * 1000;
    await new Promise(r => setTimeout(r, wait));
    return removeBackground(filePath, retries - 1);
  }

  if (!res.ok) throw new Error(await res.text());
  return Buffer.from(await res.arrayBuffer());
}
React Hook — drop-in component 
import { useState, useCallback } from 'react';

export function useBackgroundRemoval(apiKey) {
  const [state, setState] = useState({ loading: false, result: null, error: null });

  const remove = useCallback(async (file) => {
    setState({ loading: true, result: null, error: null });
    try {
      const form = new FormData();
      form.append('image_file', file);
      form.append('size', 'auto');

      const res = await fetch('https://api.remove.bg/v1.0/removebg', {
        method: 'POST',
        headers: { 'X-API-Key': apiKey ?? 'YOUR_API_KEY' },
        body: form
      });

      if (!res.ok) {
        const err = await res.json();
        throw new Error(err.errors?.[0]?.title ?? 'Unknown error');
      }

      const blob = await res.blob();
      const url = URL.createObjectURL(blob);
      setState({ loading: false, result: { url, blob }, error: null });
    } catch (e) {
      setState({ loading: false, result: null, error: e.message });
    }
  }, [apiKey]);

  return { ...state, remove };
}
